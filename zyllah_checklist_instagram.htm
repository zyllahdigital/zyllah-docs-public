var STORAGE_KEY = 'zyl_checklist_v1';

 

function salvar() {

  var estado = {};

  document.querySelectorAll('.check-item').forEach(function(item, i) {

    estado[i] = item.classList.contains('done');

  });

  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(estado)); } catch(e) {}

}

 

function carregar() {

  try {

    var raw = localStorage.getItem(STORAGE_KEY);

    if (!raw) return;

    var estado = JSON.parse(raw);

    document.querySelectorAll('.check-item').forEach(function(item, i) {

      if (estado[i]) item.classList.add('done');

    });

  } catch(e) {}

}

 

function mostrarSemana(id, btn) {

  document.querySelectorAll('.semana-panel').forEach(p => p.classList.remove('ativo'));

  document.querySelectorAll('.semana-tab').forEach(b => b.classList.remove('ativo'));

  document.getElementById(id).classList.add('ativo');

  btn.classList.add('ativo');

  atualizarProgresso(id);

}

 

function toggle(item) {

  item.classList.toggle('done');

  salvar();

  const semanaAtiva = document.querySelector('.semana-panel.ativo').id;

  atualizarProgresso(semanaAtiva);

}

 

function atualizarProgresso(semanaId) {

  const painel = document.getElementById(semanaId);

  const todos = painel.querySelectorAll('.check-item');

  const feitos = painel.querySelectorAll('.check-item.done');

  const pct = todos.length > 0 ? (feitos.length / todos.length) * 100 : 0;

  const fill = document.getElementById('prog-' + semanaId);

  const label = document.getElementById('prog-label-' + semanaId);

  if (fill) fill.style.width = pct + '%';

  if (label) label.textContent = feitos.length + ' de ' + todos.length + ' itens concluídos';

}

 

function resetarTudo() {

  if (!confirm('Apagar todos os checks e começar nova semana?')) return;

  document.querySelectorAll('.check-item').forEach(function(item) {

    item.classList.remove('done');

  });

  try { localStorage.removeItem(STORAGE_KEY); } catch(e) {}

  document.querySelectorAll('.semana-panel').forEach(function(p) {

    atualizarProgresso(p.id);

  });

}

 

// Inicializar

document.addEventListener('DOMContentLoaded', function() {

  carregar();

  document.querySelectorAll('.semana-panel').forEach(function(p) {

    atualizarProgresso(p.id);

  });

});

</script>
