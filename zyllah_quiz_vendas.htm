// ── STORAGE ──────────────────────────────────────────────────────────────────

function carregarDados() {

  try {

    const raw = localStorage.getItem('zyllah_quiz_vendas');

    return raw ? JSON.parse(raw) : { streak: 0, ultimoDia: null, melhorNota: null, sessoes: 0, temasFracos: {} };

  } catch { return { streak: 0, ultimoDia: null, melhorNota: null, sessoes: 0, temasFracos: {} }; }

}

 

function salvarDados(dados) {

  try { localStorage.setItem('zyllah_quiz_vendas', JSON.stringify(dados)); } catch {}

}

 

function atualizarStreak(dados) {

  const hoje = new Date().toDateString();

  if (dados.ultimoDia === hoje) return dados;

  const ontem = new Date(Date.now() - 86400000).toDateString();

  dados.streak = dados.ultimoDia === ontem ? (dados.streak || 0) + 1 : 1;

  dados.ultimoDia = hoje;

  return dados;

}

 

function renderizarInicio() {

  const dados = carregarDados();

  document.getElementById('streak-num').textContent = dados.streak || 0;

  document.getElementById('melhor-nota-display').textContent = dados.melhorNota !== null ? dados.melhorNota : '—';

  document.getElementById('total-sessoes').textContent = dados.sessoes || 0;

}

 

function selecionarModo(modo) {

  estado.modo = modo;

  document.querySelectorAll('.modo-btn').forEach(b => b.classList.toggle('ativo', b.dataset.modo === modo));

}

  return `Você é um treinador especialista em vendas consultivas para agências de marketing médico.

 

CONTEXTO DA EMPRESA:

${CONTEXTO_ZYLLAH}

 

Crie UMA questão de múltipla escolha estilo FGV sobre: "${tema}"

Foco: ${modoTexto}

 

FORMATO DA QUESTÃO:

- Apresente um CENÁRIO realista (o médico diz algo, ou uma situação de venda acontece)

}

 

// ── API ───────────────────────────────────────────────────────────────────────

async function carregarProximaQuestao() {

  renderizarLoading();

  try {

    const response = await fetch('https://api.anthropic.com/v1/messages', {

      method: 'POST',

      headers: { 'Content-Type': 'application/json' },

      body: JSON.stringify({

        model: 'claude-sonnet-4-20250514',

        max_tokens: 1200,

        messages: [{ role: 'user', content: gerarPrompt(estado.modo) }]

      })

    });

    if (!response.ok) throw new Error('API error');

    const data = await response.json();

    const texto = data.content[0].text.trim().replace(/```json|```/g, '').trim();

    const questao = JSON.parse(texto);

    estado.questoes.push(questao);

    renderizarQuestao(questao);

  } catch {

    renderizarErroQuestao();

  }

}

 

// ── RENDER ────────────────────────────────────────────────────────────────────

function renderizarLoading() {

  document.getElementById('quiz-corpo').innerHTML = `

    <div class="loading-pergunta">

      <div class="loading-spinner"></div>

      <div class="loading-texto">Gerando situação...</div>

    </div>`;

}

 

function renderizarErroQuestao() {

  document.getElementById('quiz-corpo').innerHTML = `

    <div class="erro-box" style="margin:0 0 20px;">Não foi possível gerar a questão. Verifique sua conexão.</div>

    <button class="btn-comecar" onclick="carregarProximaQuestao()">Tentar novamente →</button>`;

}

 

function renderizarQuestao(q) {

  const idx = estado.questaoAtual;

  document.getElementById('progresso-fill').style.width = (idx / QUESTOES_POR_SESSAO * 100) + '%';

  document.getElementById('quiz-contador').textContent = `${idx + 1} / ${QUESTOES_POR_SESSAO}`;

  renderizarVidas();

 

  const modoNome = { completo: 'Sessão Completa', reuniao: 'Script de Reunião', objecoes: 'Objeções', abordagem: 'Abordagem Inicial' }[estado.modo];

  const letras = ['A', 'B', 'C', 'D'];

  const opcoesHTML = q.alternativas.map((alt, i) => `

    <button class="opcao-btn" onclick="responder(${i})" id="opcao-${i}">

      <span class="opcao-letra">${letras[i]}</span>

      <span>${alt}</span>

    </button>`).join('');

 

  const cenarioHTML = q.cenario ? `

    <div class="cenario-box">

      <div class="cenario-label">Situação</div>

      <div class="cenario-texto">${q.cenario}</div>

    </div>` : '';

 

  document.getElementById('quiz-corpo').innerHTML = `

    <div class="quiz-modo-tag">${modoNome}</div>

    <div class="quiz-tema">${q.tema || 'Vendas'}</div>

    ${cenarioHTML}

    <div class="quiz-pergunta">${q.pergunta}</div>

    <div class="opcoes">${opcoesHTML}</div>

    <div class="feedback-box" id="feedback-box">

      <div class="feedback-titulo" id="feedback-titulo"></div>

      <div class="feedback-texto" id="feedback-texto"></div>

      <div class="feedback-principio" id="feedback-principio"></div>

    </div>

    <button class="btn-proximo" id="btn-proximo" onclick="proximaQuestao()">

      ${idx + 1 < QUESTOES_POR_SESSAO ? 'Próxima situação →' : 'Ver resultado →'}

    </button>`;

 

  estado.respondeu = false;

}

 

function renderizarVidas() {

  const container = document.getElementById('quiz-vidas');

  let html = '';

  for (let i = 0; i < VIDAS_TOTAL; i++) {

    html += `<span class="vida">${i < estado.vidas ? '❤️' : '��'}</span>`;

  }

  container.innerHTML = html;

}

 

// ── RESPONDER ─────────────────────────────────────────────────────────────────

function responder(indice) {

  if (estado.respondeu) return;

  estado.respondeu = true;

 

  const q = estado.questoes[estado.questaoAtual];

  const acertou = indice === q.correta;

 

  document.querySelectorAll('.opcao-btn').forEach(b => b.disabled = true);

  document.getElementById('opcao-' + indice).classList.add(acertou ? 'correta' : 'errada');

  if (!acertou) document.getElementById('opcao-' + q.correta).classList.add('revelada');

 

  estado.respostas.push({ acertou, tema: q.tema });

  if (!acertou) {

    estado.errosPorTema[q.tema] = (estado.errosPorTema[q.tema] || 0) + 1;

    estado.vidas--;

    renderizarVidas();

  }

 

  const fb = document.getElementById('feedback-box');

  fb.className = 'feedback-box ' + (acertou ? 'certo' : 'errado');

  fb.style.display = 'block';

  document.getElementById('feedback-titulo').textContent = acertou ? '✓ Decisão correta' : '✗ Há uma opção melhor';

  document.getElementById('feedback-texto').textContent = q.explicacao;

  document.getElementById('feedback-principio').textContent = q.principio ? '↑ ' + q.principio : '';

  document.getElementById('btn-proximo').style.display = 'block';

 

  if (estado.vidas <= 0 && !acertou) setTimeout(() => encerrarSessao(true), 1500);

}

 

async function proximaQuestao() {

  estado.questaoAtual++;

  if (estado.questaoAtual >= QUESTOES_POR_SESSAO || estado.vidas <= 0) {

    encerrarSessao(false);

  } else {

    await carregarProximaQuestao();

  }

}

 

// ── RESULTADO ─────────────────────────────────────────────────────────────────

function encerrarSessao(semVidas) {

  const acertos = estado.respostas.filter(r => r.acertou).length;

  const nota = Math.round((acertos / QUESTOES_POR_SESSAO) * 100);

 

  let dados = carregarDados();

  dados = atualizarStreak(dados);

  dados.sessoes = (dados.sessoes || 0) + 1;

  if (dados.melhorNota === null || nota > dados.melhorNota) dados.melhorNota = nota;

  for (const [tema, erros] of Object.entries(estado.errosPorTema)) {

    dados.temasFracos[tema] = (dados.temasFracos[tema] || 0) + erros;

  }

  salvarDados(dados);

 

  mostrarTela('resultado');

 

  const msgs = [

    [90, 'Domínio <em>completo</em>', 'Você está pronto para a reunião real. Confie no processo.'],

    [70, 'Bom <em>instinto</em>', 'Metodologia bem assimilada. Revise os pontos fracos abaixo.'],

    [50, 'Em <em>construção</em>', 'Continue praticando. Os padrões de venda levam repetição para fixar.'],

    [0, 'Precisa <em>revisar</em>', 'Releia o script de reunião e o banco de objeções antes da próxima sessão.']

  ];

 

  const [, titulo, sub] = msgs.find(([min]) => nota >= min) || msgs[msgs.length - 1];

  document.getElementById('resultado-nota').textContent = nota;

  document.getElementById('resultado-msg').innerHTML = titulo;

  document.getElementById('resultado-sub').textContent = sub + (semVidas ? ' (sessão encerrada por falta de vidas)' : '');

 

  const pf = document.getElementById('pontos-fracos-box');

  const temasFracos = Object.entries(dados.temasFracos).sort((a, b) => b[1] - a[1]).slice(0, 5);

  if (temasFracos.length > 0) {

    pf.innerHTML = `<div class="pf-titulo">Situações a reforçar</div>` +

      temasFracos.map(([tema, erros]) => {

        const dots = [0,1,2,3,4].map(i => `<div class="pf-dot ${i < Math.min(erros,5) ? 'erro' : ''}"></div>`).join('');

        return `<div class="pf-item"><span class="pf-tema">${tema}</span><div class="pf-barra">${dots}</div></div>`;

      }).join('');

  } else {

    pf.style.display = 'none';

  }

}

 

// ── NAV ────────────────────────────────────────────────────────────────────────

  document.getElementById('tela-quiz').style.display = tela === 'quiz' ? 'flex' : 'none';

  document.getElementById('tela-resultado').style.display = tela === 'resultado' ? 'flex' : 'none';

}

  mostrarTela('inicio');
